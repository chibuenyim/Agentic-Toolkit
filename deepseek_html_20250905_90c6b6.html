Integration with CLI, Cline, and VS Code Cursor

The Agentic Toolkit is designed to work seamlessly across different environments.

1 Global CLI Installation

Installation: The toolkit can be installed globally via npm.

npm install -g agentic-toolkit

Usage: Once installed, you can use it from any directory.

# Initialize a new project
agentic init my-project

# Generate tasks from PRD
agentic plan

# Analyze task complexity
agentic analyze

2 VS Code Cursor Integration

Setup: The toolkit includes a VS Code extension for seamless integration.

// .vscode/settings.json
{
  "agentic.enable": true,
  "agentic.autoPlan": true,
  "agentic.provider": "openrouter",
  "agentic.model": "anthropic/claude-3.5-sonnet"
}

Usage: Right-click on a PRD file and select "Generate Tasks" or use the command palette.

// Command palette options
Agentic: Plan Project
Agentic: Analyze Complexity
Agentic: Show Next Task
Agentic: Auto Run Mode

3 Cline Integration

Setup: The toolkit provides an adapter for Cline integration.

// cline integration example
const { ClineIntegration } = require('agentic-toolkit/lib/cline-adapter');

const cline = new ClineIntegration();
cline.setup({
  apiKey: process.env.OPENROUTER_API_KEY,
  model: 'anthropic/claude-3.5-sonnet'
});

Usage: Cline can call Agentic Toolkit functions directly.

// Example Cline command
/user: I want to build a notes app with authentication
/cline: @agentic generate-prd "notes app with authentication"
/agentic: ✅ PRD generated. Use `agentic plan` to create tasks.

bin/agentic-toolkit.js

#!/usr/bin/env node

const { program } = require('commander');
const { initProject } = require('../lib/project-initializer');
const { generatePRD } = require('../lib/prd-generator');
const { planTasks } = require('../lib/task-parser');
const { analyzeComplexity } = require('../lib/complexity-analyzer');
const { listTasks, getNextTask } = require('../lib/task-manager');

program
  .version('2.0.0')
  .description('Agentic Toolkit - AI-powered task management system');

program
  .command('init ')
  .description('Initialize a new project')
  .action((projectName) => {
    initProject(projectName);
  });

program
  .command('prd ')
  .description('Generate PRD from description')
  .action((description) => {
    generatePRD(description);
  });

program
  .command('plan')
  .description('Generate tasks from PRD')
  .action(() => {
    planTasks();
  });

program
  .command('analyze')
  .description('Analyze task complexity')
  .action(() => {
    analyzeComplexity();
  });

program
  .command('list')
  .description('List all tasks')
  .action(() => {
    listTasks();
  });

program
  .command('next')
  .description('Show the next task to complete')
  .action(() => {
    getNextTask();
  });

program.parse(process.argv);

lib/vscode-integration.js

const vscode = require('vscode');
const { planTasks } = require('./task-parser');
const { analyzeComplexity } = require('./complexity-analyzer');

class VSCodeIntegration {
  constructor() {
    this.statusBarItem = vscode.window.createStatusBarItem(
      vscode.StatusBarAlignment.Right, 100
    );
    this.statusBarItem.text = 'Agentic';
    this.statusBarItem.tooltip = 'Agentic Toolkit';
    this.statusBarItem.show();
  }

  activate(context) {
    // Register commands
    const planCommand = vscode.commands.registerCommand(
      'agentic.plan', 
      () => this.handlePlanCommand()
    );
    
    const analyzeCommand = vscode.commands.registerCommand(
      'agentic.analyze',
      () => this.handleAnalyzeCommand()
    );
    
    const nextTaskCommand = vscode.commands.registerCommand(
      'agentic.nextTask',
      () => this.handleNextTaskCommand()
    );

    context.subscriptions.push(planCommand, analyzeCommand, nextTaskCommand);
  }

  async handlePlanCommand() {
    const document = vscode.window.activeTextEditor?.document;
    if (document && document.languageId === 'markdown') {
      const prdContent = document.getText();
      try {
        const tasks = await planTasks(prdContent);
        vscode.window.showInformationMessage(
          `Generated ${tasks.length} tasks from PRD`
        );
      } catch (error) {
        vscode.window.showErrorMessage(
          `Failed to generate tasks: ${error.message}`
        );
      }
    }
  }

  async handleAnalyzeCommand() {
    try {
      const complexity = await analyzeComplexity();
      vscode.window.showInformationMessage(
        `Task complexity analyzed: ${complexity.score}/5`
      );
    } catch (error) {
      vscode.window.showErrorMessage(
        `Failed to analyze complexity: ${error.message}`
      );
    }
  }

  async handleNextTaskCommand() {
    try {
      const nextTask = await getNextTask();
      vscode.window.showInformationMessage(
        `Next task: ${nextTask.title} (Priority: ${nextTask.priority})`
      );
    } catch (error) {
      vscode.window.showErrorMessage(
        `Failed to get next task: ${error.message}`
      );
    }
  }
}

module.exports = { VSCodeIntegration };
